server {
    server_name umairmalik.space;

    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/umairmalik.space/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/umairmalik.space/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

}

server {
    if ($host = umairmalik.space) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    listen 80;
    server_name umairmalik.space;
    return 404; # managed by Certbot


}

server {
    listen 443 ssl;
    server_name *.umairmalik.space;

    location / {
        proxy_pass http://127.0.0.1:8081;   
        proxy_http_version 1.1;

        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # WebSockets (VERY IMPORTANT for xterm.js)
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
    }

    ssl_certificate /etc/letsencrypt/live/umairmalik.space/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/umairmalik.space/privkey.pem;
}
